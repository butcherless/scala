Obtener dependencias mediante comandos bash

 gw dep | \
 grep -e '([0-9a-z\.-]*):([a-z-]*):([a-zA-Z0-9\.-]*)' | \
 sed 's/[+\]---//g; s/|//g' | \
 awk '{print $1}' | \
 sort -u
